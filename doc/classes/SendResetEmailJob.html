def perform(candidate, type)
  case type
    when AdminsController::RESET_PASSWORD
      raise "Do not do SendResetEmailJob type '#{type}'"
    when AdminsController::INITIAL_EMAIL
      raise "Do not do SendResetEmailJob type '#{type}'"

  end
